import pandas as pd
import matplotlib.pyplot as plt

# Load CSVs into DataFrames
brand_df = pd.read_csv("brand_category.csv")
category_df = pd.read_csv("categories.csv")
offer_df = pd.read_csv("offer_retailer.csv")

# brandCategory.csv analysis

# Top 10 brands by receipts  
top_brands = brand_df.sort_values('RECEIPTS', ascending=False).head(10)
print(top_brands)

# Visualize top brand receipts
ax = top_brands['RECEIPTS'].plot.bar(title='Top 10 Brands by Receipts')
ax.set_ylabel('Receipts')
plt.tight_layout()
plt.show()

# Categories for top brands 
print(top_brands['BRAND_BELONGS_TO_CATEGORY'].value_counts())

# categories.csv analysis

# Count child categories per parent
parent_counts = category_df.groupby('IS_CHILD_CATEGORY_TO').size()
print(parent_counts)

# Plot hierarchy 
parent_counts.plot.bar(title='Child Categories per Parent')
plt.tight_layout()
plt.show()

# offerRetailer.csv analysis

# Count offers by retailer  
print(offer_df['RETAILER'].value_counts())

# Count offers by brand
print(offer_df['BRAND'].value_counts())
